#include <iostream>
using namespace std;

int main()
{
    int choice;

    cout << "=========================================\n";
    cout << "        WEEK 1 - BASICS OF C++\n";
    cout << "=========================================\n";

    cout << "\nTopics Covered:\n";
    cout << "1. Syntax, Variables, Data Types\n";
    cout << "2. Input/Output & Operators\n";
    cout << "3. Loops and Conditional Statements\n";

    cout << "\n=========== ASSIGNMENTS ===========\n";
    cout << "1. Simple Calculator\n";
    cout << "2. Sum and Average of Array Elements\n";
    cout << "3. Check Prime Number\n";
    cout << "4. Reverse a Number\n";
    cout << "5. Mini Project - ATM Menu Program\n";

    cout << "\nEnter your choice (1-5): ";
    cin >> choice;

    switch(choice)
    {
        //=========================================
        // Assignment 1 - Simple Calculator
        //=========================================
        case 1:
        {
            double a, b;
            char op;

            cout << "\n----- Assignment 1: Simple Calculator -----\n";
            cout << "Enter first number: ";
            cin >> a;

            cout << "Enter operator (+,-,*,/): ";
            cin >> op;

            cout << "Enter second number: ";
            cin >> b;

            switch(op)
            {
                case '+':
                    cout << "Result = " << a + b;
                    break;

                case '-':
                    cout << "Result = " << a - b;
                    break;

                case '*':
                    cout << "Result = " << a * b;
                    break;

                case '/':
                    if(b != 0)
                        cout << "Result = " << a / b;
                    else
                        cout << "Division by zero is not allowed.";
                    break;

                default:
                    cout << "Invalid Operator.";
            }
            break;
        }

        //=========================================
        // Assignment 2 - Sum and Average
        //=========================================
        case 2:
        {
            cout << "\n----- Assignment 2: Sum and Average of Array -----\n";

            int n;
            cout << "Enter number of elements: ";
            cin >> n;

            int arr[100];
            int sum = 0;

            cout << "Enter elements:\n";

            for(int i = 0; i < n; i++)
            {
                cin >> arr[i];
                sum += arr[i];
            }

            double average = (double)sum / n;

            cout << "Sum = " << sum << endl;
            cout << "Average = " << average;

            break;
        }

        //=========================================
        // Assignment 3 - Prime Number
        //=========================================
        case 3:
        {
            cout << "\n----- Assignment 3: Prime Number Check -----\n";

            int num;
            bool prime = true;

            cout << "Enter a number: ";
            cin >> num;

            if(num <= 1)
                prime = false;

            for(int i = 2; i <= num/2; i++)
            {
                if(num % i == 0)
                {
                    prime = false;
                    break;
                }
            }

            if(prime)
                cout << num << " is a Prime Number.";
            else
                cout << num << " is NOT a Prime Number.";

            break;
        }

        //=========================================
        // Assignment 4 - Reverse Number
        //=========================================
        case 4:
        {
            cout << "\n----- Assignment 4: Reverse a Number -----\n";

            int num, reverse = 0, rem;

            cout << "Enter a number: ";
            cin >> num;

            while(num != 0)
            {
                rem = num % 10;
                reverse = reverse * 10 + rem;
                num /= 10;
            }

            cout << "Reversed Number = " << reverse;

            break;
        }

        //=========================================
        // Mini Project - ATM Menu
        //=========================================
        case 5:
        {
            cout << "\n----- MINI PROJECT : ATM MENU PROGRAM -----\n";

            double balance = 10000;
            int option;
            double amount;

            do
            {
                cout << "\n========== ATM MENU ==========\n";
                cout << "1. Check Balance\n";
                cout << "2. Deposit Money\n";
                cout << "3. Withdraw Money\n";
                cout << "4. Exit\n";

                cout << "Enter your option: ";
                cin >> option;

                switch(option)
                {
                    case 1:
                        cout << "Current Balance = Rs. " << balance << endl;
                        break;

                    case 2:
                        cout << "Enter Deposit Amount: ";
                        cin >> amount;
                        balance += amount;
                        cout << "Amount Deposited Successfully.\n";
                        break;

                    case 3:
                        cout << "Enter Withdraw Amount: ";
                        cin >> amount;

                        if(amount <= balance)
                        {
                            balance -= amount;
                            cout << "Withdrawal Successful.\n";
                        }
                        else
                        {
                            cout << "Insufficient Balance.\n";
                        }
                        break;

                    case 4:
                        cout << "Thank You for Using ATM.\n";
                        break;

                    default:
                        cout << "Invalid Option.\n";
                }

            } while(option != 4);

            break;
        }

        default:
            cout << "Invalid Choice!";
    }

    return 0;
}
